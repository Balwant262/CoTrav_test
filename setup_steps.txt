deactivate
sudo nano /etc/systemd/system/gunicorn.service - add the below module in config file
-------------------------------------
[Unit]
Description=gunicorn daemon
After=network.target

[Service]
User=root
Group=nginx
WorkingDirectory=/home/CoTrav
ExecStart=/home/CoTrav/myprojectenv/bin/gunicorn --workers 3 --bind unix:/home/CoTrav/CoTrav.sock CoTrav.wsgi:application

[Install]
WantedBy=multi-user.target
-------------------------------------------
sudo systemctl start gunicorn
sudo systemctl enable gunicorn
sudo nano /etc/nginx/nginx.conf - add the below server module in the config file
-------------------------------
server {
    listen 80;
    server_name 159.89.174.70;

    location = /favicon.ico { access_log off; log_not_found off; }
    location /static/ {
        root /home/CoTrav;
    }

    location / {
        proxy_set_header Host $http_host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_pass http://unix:/home/CoTrav/CoTrav.sock;
    }
}
------------------------------------
